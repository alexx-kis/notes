# файлы ресурсов

- теперь научимся отдавать браузеру файлы ресурсов (картинки, файлы css стилей, клиентских скриптов итд)

- рассмотрим работу с ресурсами на примере картинок
- пусть по адресу '/page.html' мы хотим отдавать HTML-страницу, а по адресу '/image.png' - картинку

- для больших картинок нужно использовать поток

```js
http
  .createServer(async (request, response) => {
    if (request.url !== '/favicon.ico') {
      let status = 200;

      if (request.url === '/page.html') {
        const data = await fsp.readFile('page.html', 'utf8');

        response.writeHead(status, { 'Content-Type': 'text/html' });
        response.end(data);

        return;
      }

      if (request.url === '/image.png') {
        response.writeHead(status, { 'Content-Type': 'image/png' });

        const stream = fs.createReadStream('image.png');
        stream.pipe(response);

        return;
      }

      response.writeHead(404, { 'Content-Type': 'text/plain' });
      response.end('Not found');
    }
  })
  .listen(3000);
```

## автоматические запросы браузера

- рассмотрим HTML-страницу:

```html
<!DOCTYPE html>
<html>
  <head>
    <meta charset="utf-8" />
    <title>my page</title>
    <link rel="stylesheet" href="styles.css" />
    <script src="script.js"></script>
  </head>
  <body>
    <p>my first page on NodeJS</p>
    <img src="img.png" />
  </body>
</html>
```

- на данной странице подключается css-файл, js-файл и картинка
- это значит, что, когда браузер получит и прочитает HTML-код страницы, он автоматически отправит серверу ещё 3 запроса для получения данных файлов

```js
http
  .createServer(async (request, response) => {
    let type = 'text/html';
    let status = 200;

    if (request.url === '/favicon.ico') {
      const favicon = fs.readFile('favicon.ico');
      response.end(favicon);
    }

    if (request.url === '/page.html') {
      const data = await fsp.readFile('page.html', 'utf8');
      response.writeHead(status, { 'Content-Type': type });
      response.end(data);
      return;
    }

    if (request.url === '/styles.css') {
      const data = await fsp.readFile('styles.css', 'utf8');
      response.end(data);
      return;
    }

    if (request.url === '/script.js') {
      const data = await fsp.readFile('script.js', 'utf8');
      response.end(data);
      return;
    }

    if (request.url === '/image.png') {
      type = 'image/png';

      response.writeHead(status, { 'Content-Type': type });

      const stream = fs.createReadStream('image.png');
      stream.pipe(response);

      return;
    }

    response.writeHead(404, { 'Content-Type': 'text/plain' });
    response.end('Not found');
  })
  .listen(3000);
```
