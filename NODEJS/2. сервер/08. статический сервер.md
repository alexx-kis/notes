# статический сервер

- теперь сделаем так, чтобы по URL на сайте искался соответствующий ему HTML-файл
- например, если запрашивается '/page.html', то мы должны отдать такой же файл, а если запрашивается 'dir/test.html', то мы должны отдать файл test.html из папки dir

- разместим все HTML-файлы в папке root
- это будет корневая папка сайта и поиск HTML-файлов будем начинать относительно этой папки

```js
http
  .createServer(async (request, response) => {
    if (request.url === '/favicon.ico') {
      const path = 'root' + request.url;
      const text = await fsp.readFile(path, 'utf8');

      response.writeHead(200, { 'Content-type': 'text/html' });
      response.write(text);
      response.end();
    }
  })
  .listen(3000);
```

- теперь обработаем URL вида '/dir/sub/'
- в этом адресе не указано имя файла и его расширение
- принято считать, что такой адрес ссылается на файл `index.html`, находящийся в этой папке
- то есть наш URL следует трактовать как '/dir/sub/index.html'
- модифицируем код:

```js
http
  .createServer(async (request, response) => {
    if (request.url === '/favicon.ico') {
      let path = 'root' + request.url;

      if ((await fsp.stat(path)).isDirectory()) {
        path += 'index.html';
      }

      const text = await fsp.readFile(path, 'utf8');

      response.writeHead(200, { 'Content-type': 'text/html' });
      response.write(text);
      response.end();
    }
  })
  .listen(3000);
```

- если запрошенный URL не соответствует файлу на нашем сайте, мы должны отдавать 404 ошибку:

```js
http
  .createServer(async (request, response) => {
    if (request.url !== '/favicon.ico') {
      let text;
      let status;
      let path = 'root' + request.url;

      try {
        if ((await fsp.stat(path)).isDirectory()) {
          path += '/index.html';
        }

        status = 200;
        text = await fsp.readFile(path, 'utf8');
      } catch (error) {
        status = 404;
        text = await fsp.readFile('root/not-found.html', 'utf8');
        console.log(error);
      }

      response.writeHead(status, { 'Content-Type': 'text/html' });
      response.write(text);
      response.end();
    }
  })
  .listen(3000);
```

структура сайта:

```
example-dir
  |- root
  |   |- index.html
  |   |- not-found.html
  index.js
```

- в корневой папке 'example-dir' находится папка 'root' внутри которой html-страницы, а файл сервера index.js лежит в корневой папке

## выдача ресурсов

- теперь модифицируем сервер так, чтобы кроме HTML-файлов автоматически так же выдавались запрошенные ресурсы
- создадим функцию, которая будет принимать пусть к файлу и по расширению этого файла выдавать его mime тип:

```js
const getMimeType = (path) => {
  const mimes = {
    html: 'text/html',
    jpeg: 'image/jpeg',
    jpg: 'image/jpeg',
    png: 'image/png',
    svg: 'image/svg',
    ico: 'image/x-icon',
    json: 'application/json',
    js: 'text/javascript',
    css: 'text/css',
  };

  const extensions = Object.keys(mimes);
  const extensionRegExp = new RegExp('\\.(' + extensions.join('|') + ')$');

  const extension = path.match(extensionRegExp)[1];

  if (extension) {
    return mimes[extension];
  } else {
    return 'text/plain';
  }
};
```

- имея такую функцию, легко адаптировать сервер для выдачи файлов любого типа:

```js
http
  .createServer(async (request, response) => {
    let text;
    let status;
    let path = 'root' + request.url;

    try {
      if ((await fsp.stat(path)).isDirectory()) {
        path += '/index.html';
      }
      status = 200;
      text = await fsp.readFile(path, 'utf8');
    } catch (error) {
      status = 404;
      text = 'page not found';
    }

    response.writeHead(status, { 'Content-type': getMimeType(path) });
    response.write(text);
    response.end();
  })
  .listen(3000);
```

## убираем расширение из URL

- мы реализовали сервер так, что все URL (кроме адресов папок) заканчиваются расширением `.html`
- однако в современном мире наличие расширения файла в адресе недопустимо

- необходимо сделать так, чтобы адрес вида `/page` соответствовал файлу `root/page.html`, адрес вида `/dir/page` - файлу `root/dir/page/page.html`, а адрес главной страницы `/` - файлу `root/index.html`

```js
http
  .createServer(async (request, response) => {
    let text;
    let status;
    let path = 'root' + request.url;

    console.log(path);

    try {
      if (!path.includes('.') && request.url !== '/') {
        path += '.html';
      }
      if ((await fsp.stat(path)).isDirectory()) {
        path += '/index.html';
      }
      status = 200;
      text = await fsp.readFile(path, 'utf8');
    } catch (error) {
      status = 404;
      text = 'page not found';
    }

    response.writeHead(status, { 'Content-type': getMimeType(path) });
    response.write(text);
    response.end();
  })
  .listen(3000);
```
