# простой роутинг

- теперь будем выдавать разное содержимое при обращении к разным URL:

```js
http
  .createServer((request, response) => {
    if (request.url !== '/favicon.ico') {
      if (request.url === '/page1') {
        response.writeHead(200, { 'content-type': 'text/html' });
        response.write('page 1');
        response.end();
      }
      if (request.url === '/page2') {
        response.writeHead(200, { 'content-type': 'text/html' });
        response.write('page 2');
        response.end();
      }
      if (request.url === '/page3') {
        response.writeHead(200, { 'content-type': 'text/html' });
        response.write('page 3');
        response.end();
      }
    }
  })
  .listen(3000);
```

- часть кода дублируется, поэтому упростим:

```js
http
  .createServer((request, response) => {
    if (request.url !== '/favicon.ico') {
      let text;

      if (request.url === '/page1') {
        text = 'page 1';
      }
      if (request.url === '/page2') {
        text = 'page 2';
      }
      if (request.url === '/page3') {
        text = 'page 3';
      }

      response.writeHead(200, { 'content-type': 'text/html' });
      response.write(text);
      response.end();
    }
  })
  .listen(3000);
```

- сделаем так, чтобы при обращении к любому другому адресу выдавалось сообщение о том, что страница не найдена:

```js
http
  .createServer((request, response) => {
    if (request.url !== '/favicon.ico') {
      let text;

      if (request.url === '/page1') {
        text = 'page 1';
      } else if (request.url === '/page2') {
        text = 'page 2';
      } else if (request.url === '/page3') {
        text = 'page 3';
      } else {
        text = 'page not found';
      }

      response.writeHead(200, { 'content-type': 'text/html' });
      response.write(text);
      response.end();
    }
  })
  .listen(3000);
```

- чтобы браузер и поисковые системы правильно обрабатывали отсутствие страницы, нужно для существующих страниц отдавать статус 200, а для не существующих - 404:

```js
http
  .createServer((request, response) => {
    if (request.url !== '/favicon.ico') {
      let text;
      let status;

      if (request.url === '/page1') {
        text = 'page 1';
        status = 200;
      } else if (request.url === '/page2') {
        text = 'page 2';
        status = 200;
      } else if (request.url === '/page3') {
        text = 'page 3';
        status = 200;
      } else {
        text = 'page not found';
        status = 404;
      }

      response.writeHead(status, { 'content-type': 'text/html' });
      response.write(text);
      response.end();
    }
  })
  .listen(3000);
```

- можно упростить - объявить статус 200 изначально и менять на 404 только для несуществующей страницы:

```js
http
  .createServer((request, response) => {
    if (request.url !== '/favicon.ico') {
      let text;
      let status = 200;

      if (request.url === '/page1') {
        text = 'page 1';
      } else if (request.url === '/page2') {
        text = 'page 2';
      } else if (request.url === '/page3') {
        text = 'page 3';
      } else {
        text = 'page not found';
        status = 404;
      }

      response.writeHead(status, { 'content-type': 'text/html' });
      response.write(text);
      response.end();
    }
  })
  .listen(3000);
```

- можно переписать с помощью конструкции `switch-case`:

```js
http
  .createServer((request, response) => {
    if (request.url !== '/favicon.ico') {
      let text;
      let status = 200;

      switch (request.url) {
        case '/page1':
          text = 'page1';
          break;
        case '/page2':
          text = 'page2';
          break;
        case '/page3':
          text = 'page3';
          break;
        default:
          text = 'page not found';
          status = 404;
          break;
      }

      response.writeHead(status, { 'content-type': 'text/html' });
      response.write(text);
      response.end();
    }
  })
  .listen(3000);
```

### задача

- Дан объект с URL-ами и соответствующими им текстами страниц:

```js
let obj = {
  '/page1': '1',
  '/page2': '2',
  '/page3': '3',
};
```

- Сделайте сервер на основе этого объекта. При запросе существующего в объекте адреса отдавайте соответствующий текст, а при запросе отсутствующего - сообщение об ошибке и статус 404.

### решение

```js
const obj = {
  '/page1': '1',
  '/page2': '2',
  '/page3': '3',
};

http
  .createServer((request, response) => {
    if (request.url !== '/favicon.ico') {
      let text;
      let status = 200;

      if (obj[request.url]) {
        text = obj[request.url];
      } else {
        text = 'page not found';
        status = 404;
      }

      response.writeHead(status, { 'content-type': 'text/html' });
      response.write(text);
      response.end();
    }
  })
  .listen(3000);
```
